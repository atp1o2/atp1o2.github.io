Include method

.indexOf as a conditional
arr.indexOf(element) !== -1

If item is not in the array, -1 is returned.

Can be used on an Array or String

Example:

var arr = [1,2,"apple"];

if (arr.indexOf("apple") !== -1){
  console.log("It is inside of the array")
}
else{
 console.log("It's not!")
}
// => returns "It is inside of the array"
function include(element, arr)

Use a function that accepts an element and checks if array contains the element
returns true/false

function include(element, arr) {
  for (var i = 0; i < arr.length; i++) {
   return (arr[i] === element) ? true : false;
  }
}

// for max integer

function maxValue(obj){
 var max = 0
 for(var x in obj) {
  if( obj[x] > max){
   max = obj[x]
  }
 }
 return max
}


// for max string length

function maxValue(obj){
 var max = 0, ans = ""
 for(var x in obj) {
  if( obj[x].length > max){
   max = obj[x].length
   ans = obj[x];
  }
 }
 return ans
}
